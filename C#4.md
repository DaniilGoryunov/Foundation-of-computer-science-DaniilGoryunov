#include <stdio.h>
int main() {
    int k;
    printf("Введите число: ");
    scanf("%d",&k);
    int j=1,n=0;
    while (j/k==0) {
        n++;
        j*=10;
    }
    int a[n],i;
    for (i=1;i<=n;i++){
        a[n-i]=k%10;
        k/=10;
    }
    int z;
    for (z=0;z<n-2;z++){
        if (a[z]==a[z+1]+a[z+2]){
            printf("%d\n",a[z]);
        }
    }
    return 0;
}
